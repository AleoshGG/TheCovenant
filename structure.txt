mi-juego/
├── main.go               # Punto de entrada: Inicia el juego y la ventana.
├── go.mod
├── go.sum
│
├── game/                 # El paquete 'game' contiene el struct principal 'Game'.
│   └── game.go           # Aquí defines tu struct 'Game' que implementa
│                         # la interfaz ebiten.Game (Update, Draw, Layout).
│
├── scenes/               # Gestión de las escenas o estados del juego.
│   ├── scene.go          # Define la interfaz 'Scene' (ej. Update(), Draw()).
│   ├── manager.go        # Un 'SceneManager' que maneja la escena activa.
│   ├── title_scene.go    # Implementación de la escena del menú.
│   └── play_scene.go     # Implementación de la escena principal del juego.
│
├── entities/             # Todos los objetos/entidades del juego.
│   ├── player.go         # Struct y lógica del jugador.
│   ├── enemy.go          # Struct y lógica de los enemigos.
│   └── bullet.go         # Struct y lógica de las balas.
│
├── assets/               # Paquete de Go para *cargar* y *gestionar* assets.
│   ├── images.go         # Funciones para cargar sprites (ej. LoadPNG).
│   ├── audio.go          # Funciones para cargar sonidos y música.
│   └── fonts.go          # Funciones para cargar fuentes.
│
├── input/                # (Opcional) Abstracción de la entrada del usuario.
│   └── input.go          # Simplifica la gestión de teclas/gamepad.
│
├── config/               # Constantes y configuración.
│   └── config.go         # Ej. ScreenWidth, ScreenHeight, Title, DebugMode.
│
└── resources/            # Los archivos de assets físicos.
    ├── images/
    │   ├── player.png
    │   └── tileset.png
    ├── audio/
    │   ├── jump.wav
    │   └── music.ogg
    └── fonts/
        └── pixel_font.ttf
